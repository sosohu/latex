\begin{description}
    \item{\textbf{问题}}: Given a list, return the last Kth Node.
    \item{\textbf{迭代}} : \fbox{时间复杂度O(n), 空间复杂度O(1)}
	两个指针问题.
    \begin{lstlisting}
ListNode * RGetKthNode(ListNode *pHead, unsigned int k){
	if(k < 1)	return NULL;
	ListNode *pre = pHead;
	ListNode *last = pHead;
	while(k-- > 0 && last){
		last = last->next;
	}
	if(!last) return NULL;
	while(last){
		last = last->next;
		pre = pre->next;
	}
	return pre;	
}
    \end{lstlisting}
\end{description}
