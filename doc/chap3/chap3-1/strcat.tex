\begin{description}
    \item{\textbf{问题}}: 将一个字符串连接到另一个字符串后面
    \item{\textbf{Care}} : \fbox{时间复杂度O(n), 空间复杂度O(1)}
	\\输入参数const性和非NULL,尾赋$'\backslash0'$, 有返回值
    \begin{lstlisting}
char *strcat(char *strDest, const char *strScr){
	assert(strDest && strScr);
	if(!strScr)	return strDest;
	char* p = strDest;
	while(*p){
		p++;
	}
	while(*strScr){
		*p++ = *strScr++;
	}
	*p = '\0';
	return strDest;
}
    \end{lstlisting}
\end{description}
